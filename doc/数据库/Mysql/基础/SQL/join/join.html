Join(иҝһжҺҘ) вҖӢ

жҰӮеҝө вҖӢ

  • SQL Join иҝһжҺҘеӯҗеҸҘз”ЁдәҺеңЁдёӨдёӘжҲ–жӣҙеӨҡеңЁж•°жҚ®еә“дёӯзҡ„иЎЁзҡ„и®°еҪ•з»„еҗҲгҖӮ

  • JOINжҳҜйҖҡиҝҮдҪҝз”Ёд»ҺдёӨдёӘиЎЁеӯ—ж®өе…ұеҗҢзҡ„еҖјз»„еҗҲиҝһжҺҘиө·жқҘгҖӮ

  • MySQLе®ҳж–№еҸӘжҸҗдҫӣдәҶдёүз§Қjoinж–№ејҸпјҢеҶ…иҝһжҺҘгҖҒе·ҰиҝһжҺҘе’ҢеҸіиҝһжҺҘпјҢдёҚж”ҜжҢҒе…¶д»–зҡ„иҝһжҺҘе…ій”®еӯ—гҖӮ

  • еҸҜд»ҘйҖҡиҝҮдёҖе®ҡзҡ„иҜӯжі•е°ҶиҫҫеҲ°е…¶д»–зҡ„иҝһжҺҘзҡ„ж•ҲжһңгҖӮ

  • е…ій”®еӯ—пјҡinner join + on
  • иҜҙжҳҺпјҡinner join е…Ёз§°жҳҜеҶ…иҝһжҺҘгҖӮеҸӘжҹҘиҜўеҗ„дёӘиЎЁеҢ№й…Қзҡ„и®°еҪ•гҖӮ

е·®йӣҶпјҲMinusпјү вҖӢ

  • е…ій”®еӯ—пјҡleft join + on + where !=
  • иҜҙжҳҺпјҡ дёӨиЎЁзҡ„е…ЁиҝһжҺҘдёӯйҷӨеҺ»йҮҚеҗҲзҡ„йғЁеҲҶпјҢеҚідёӨеј иЎЁеҲҶеҲ«зҡ„зү№жңүйғЁеҲҶзҡ„еҗҲйӣҶгҖӮ
sql
drop table t1;
drop table t2;
create table t1(id int primary key,nickname varchar(20),playNum varchar(20));
create table t2(id int primary key,nickname varchar(20),playNum varchar(20));
insert into t1 values(1,1,10);
insert into t1 values(2,2,20);
insert into t1 values(3,3,30);
insert into t2 values(1,1,10);
insert into t2 values(2,2,200);
insert into t2 values(3,33,300);
commit;

SELECT t1.id, t1.nickname, t1.playNum
FROM t1 LEFT JOIN t2 ON t1.id = t2.id
WHERE t1.nickname != t2.nickname
    OR t1.playNum != t2.playNum;
